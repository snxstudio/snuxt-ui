## Usage

```tsx
import { SuggestionChips, SuggestionChip } from '@snuxt-ui/react'
```

### Variants

```tsx
<SuggestionChips>
  <SuggestionChip variant="default">Default</SuggestionChip>
  <SuggestionChip variant="primary">Primary</SuggestionChip>
  <SuggestionChip variant="outline">Outline</SuggestionChip>
</SuggestionChips>
```

### With Click Handler

```tsx
<SuggestionChips>
  <SuggestionChip onClick={() => send('Tell me more')}>
    Tell me more
  </SuggestionChip>
</SuggestionChips>
```
