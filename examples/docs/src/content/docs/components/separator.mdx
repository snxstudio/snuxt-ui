## Usage

```tsx
import { Separator } from '@snuxt-ui/react'
```

### Horizontal

```tsx
<div>
  <p>Above</p>
  <Separator />
  <p>Below</p>
</div>
```

### Vertical

```tsx
<div className="flex items-center gap-3">
  <span>Left</span>
  <Separator orientation="vertical" className="h-4" />
  <span>Right</span>
</div>
```

### Between List Items

```tsx
<div className="space-y-2">
  <p>Item one</p>
  <Separator />
  <p>Item two</p>
  <Separator />
  <p>Item three</p>
</div>
```
