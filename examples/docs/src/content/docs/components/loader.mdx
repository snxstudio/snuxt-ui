## Usage

```tsx
import { Loader } from '@snuxt-ui/react'
```

### Small

```tsx
<Loader size="sm" />
```

### Medium (Default)

```tsx
<Loader size="md" />
```

### Large

```tsx
<Loader size="lg" />
```

### Extra Large

```tsx
<Loader size="xl" />
```

### Muted Variant

```tsx
<Loader variant="muted" />
```

### Inline with Text

```tsx
<div className="flex items-center gap-2">
  <Loader size="sm" />
  <span>Loading data...</span>
</div>
```

### Full Page

```tsx
<div className="flex items-center justify-center h-64">
  <Loader size="lg" />
</div>
```
