      standalone: true,
      imports: [SnxSwitchComponent],
      template: `
        <div class="flex items-center gap-3">
          <snx-switch
            (checkedChange)="onCheckedChange($event)" />
          <span class="text-sm">{{ on ? 'On' : 'Off' }}</span>
        </div>
      `,
    })
    export class SwitchDemoComponent {
      on = false;

      onCheckedChange(checked: boolean) {
        this.on = checked;
      }
    }

## Usage

```tsx
import { Switch } from '@snuxt-ui/react'
```

### Default

```tsx
const [on, setOn] = useState(false)

<Switch checked={on} onCheckedChange={setOn} />
```

### With Label

```tsx
const [on, setOn] = useState(false)

<div className="flex items-center gap-3">
  <Switch id="airplane" checked={on} onCheckedChange={setOn} />
  <label htmlFor="airplane" className="text-sm">Airplane Mode</label>
</div>
```

### Disabled

```tsx
<Switch checked={false} disabled />
```
