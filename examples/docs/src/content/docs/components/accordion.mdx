## Usage

```tsx
import {
  Accordion, AccordionItem,
  AccordionTrigger, AccordionContent,
} from '@snuxt-ui/react'
```

### Single (Collapsible)

```tsx
const items = [{ value: 'item-1' }, { value: 'item-2' }]

<Accordion items={items} type="single" collapsible>
  {(api) => (
    <>
      <AccordionItem>
        <AccordionTrigger
          isExpanded={api.expandedValues.includes('item-1')}
          onClick={() =>
            api.getTriggerProps('item-1', api.expandedValues).onClick()
          }
        >
          Section 1
        </AccordionTrigger>
        {api.expandedValues.includes('item-1') && (
          <AccordionContent>Content for section 1.</AccordionContent>
        )}
      </AccordionItem>
      <AccordionItem>
        <AccordionTrigger
          isExpanded={api.expandedValues.includes('item-2')}
          onClick={() =>
            api.getTriggerProps('item-2', api.expandedValues).onClick()
          }
        >
          Section 2
        </AccordionTrigger>
        {api.expandedValues.includes('item-2') && (
          <AccordionContent>Content for section 2.</AccordionContent>
        )}
      </AccordionItem>
    </>
  )}
</Accordion>
```
