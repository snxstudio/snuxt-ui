## Usage

```tsx
import { Tooltip } from '@snuxt-ui/react'
```

### Basic

```tsx
<Tooltip content="Save your changes">
  <Button>Hover me</Button>
</Tooltip>
```

### Bottom Placement

```tsx
<Tooltip content="Tooltip below" side="bottom">
  <Button variant="outline">Hover me</Button>
</Tooltip>
```

### Custom Delay

```tsx
<Tooltip content="Appears after 500ms" delay={500}>
  <Button variant="outline">Slow tooltip</Button>
</Tooltip>
```

### With Custom Styling

```tsx
<Tooltip content="Styled tooltip" className="bg-blue-600 text-white">
  <Button variant="ghost">Custom style</Button>
</Tooltip>
```
