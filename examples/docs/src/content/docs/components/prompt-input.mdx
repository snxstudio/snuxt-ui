A composable prompt input for AI chat interfaces. Includes a resizable textarea, action buttons slot, character count, and file attachment pills.

    ```tsx
    import { useState } from 'react'
    import {
      PromptInput, PromptInputTextarea, PromptInputFooter,
      PromptInputActions, PromptInputCharCount,
      PromptInputAttachments, PromptInputAttachment, Button,
    } from '@snuxt-ui/react'

    export function PromptInputDemo() {
      const [value, setValue] = useState('')

      return (
        <PromptInput>
          <PromptInputAttachments>
            <PromptInputAttachment onRemove={() => {}}>report.pdf</PromptInputAttachment>
          </PromptInputAttachments>
          <PromptInputTextarea
            placeholder="Ask me anything..."
            value={value}
            onChange={(e) => setValue(e.target.value)}
          />
          <PromptInputFooter>
            <PromptInputActions>
              <Button variant="primary" size="icon" disabled={!value.trim()}>Send</Button>
            </PromptInputActions>
            <PromptInputCharCount count={value.length} max={1000} />
          </PromptInputFooter>
        </PromptInput>
      )
    }
    ```

## Usage

```tsx
import {
  PromptInput, PromptInputTextarea, PromptInputFooter,
  PromptInputActions, PromptInputCharCount,
} from '@snuxt-ui/react'
```

### Basic

```tsx
<PromptInput>
  <PromptInputTextarea placeholder="Type a message..." />
  <PromptInputFooter>
    <PromptInputActions>
      <Button variant="primary" size="icon">Send</Button>
    </PromptInputActions>
    <PromptInputCharCount count={0} max={500} />
  </PromptInputFooter>
</PromptInput>
```

### With Attachments

```tsx
<PromptInput>
  <PromptInputAttachments>
    <PromptInputAttachment onRemove={() => {}}>file.pdf</PromptInputAttachment>
  </PromptInputAttachments>
  <PromptInputTextarea placeholder="Ask about the file..." />
  <PromptInputFooter>
    <PromptInputActions>
      <Button variant="ghost" size="icon">Attach</Button>
      <Button variant="primary" size="icon">Send</Button>
    </PromptInputActions>
  </PromptInputFooter>
</PromptInput>
```
